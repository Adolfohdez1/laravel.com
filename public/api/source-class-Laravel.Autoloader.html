 27:      * All of the class aliases registered with the auto-loader.
 28:      *
 29:      * @var array
 30:      */
 31:     public static $aliases = array();
 32: 
 33:     /**
 34:      * Load the file corresponding to a given class.
 35:      *
 36:      * This method is registerd in the bootstrap file as an SPL auto-loader.
 37:      *
 38:      * @param  string  $class
 39:      * @return void
 40:      */
 41:     public static function load($class)
 42:     {
 43:         // First, we will check to see if the class has been aliased. If it has,
 44:         // we will register the alias, which may cause the auto-loader to be
 45:         // called again for the "real" class name to load its file.
 46:         if (isset(static::$aliases[$class]))
 47:         {
 48:             class_alias(static::$aliases[$class], $class);
 49:         }
 50: 
 51:         // All classes in Laravel are staticly mapped. There is no crazy search
 52:         // routine that digs through directories. It's just a simple array of
 53:         // class to file path maps for ultra-fast file loading.
 54:         elseif (isset(static::$mappings[$class]))
 55:         {
 56:             require static::$mappings[$class];
 57:         }
 58: 
 59:         // If the class namespace is mapped to a directory, we will load the
 60:         // class using the PSR-0 standards from that directory accounting
 61:         // for the root of the namespace by trimming it.
 62:         if ( ! is_null($info = static::namespaced($class)))
 63:         {
 64:             $class = substr($class, strlen($info['namespace']));
 65: 
 66:             return static::load_psr($class, $info['directory']);
 67:         }
 68: 
 69:         static::load_psr($class);
 70:     }
 71: 
 72:     /**
 73:      * Attempt to resolve a class using the PSR-0 standard.
 74:      *
 75:      * @param  string  $class
 76:      * @param  string  $directory
 77:      * @return void
 78:      */
 79:     protected static function load_psr($class, $directory = null)
 80:     {
 81:         // The PSR-0 standard indicates that class namespaces and underscores
 82:         // shoould be used to indcate the directory tree in which the class
 83:         // resides, so we'll convert them to slashes.
 84:         $file = str_replace(array('\\', '_'), '/', $class);
 85: 
 86:         $directories = $directory ?: static::$directories;
 87: 
 88:         $lower = strtolower($file);
 89: 
 90:         // Once we have formatted the class name, we'll simply spin through
 91:         // the registered PSR-0 directories and attempt to locate and load
 92:         // the class file into the script.
 93:         foreach ((array) $directories as $directory)
 94:         {
 95:             if (file_exists($path = $directory.$lower.EXT))
 96:             {
 97:                 return require $path;
 98:             }
 99:             elseif (file_exists($path = $directory.$file.EXT))
100:             {
101:                 return require $path;
102:             }
103:         }
104:     }
105: 
106:     /**
107:      * Get the directory for a given namespaced class.
108:      *
109:      * @param  string  $class
110:      * @return string
111:      */
112:     protected static function namespaced($class)
113:     {
114:         foreach (static::$namespaces as $namespace => $directory)
115:         {
116:             if (starts_with($class, $namespace))
117:             {
118:                 return compact('namespace', 'directory');
119:             }
120:         }
121:     }

124:      * Register an array of class to path mappings.
125:      *
126:      * @param  array  $mappings
127:      * @return void
128:      */
129:     public static function map($mappings)
130:     {
131:         static::$mappings = array_merge(static::$mappings, $mappings);
132:     }

135:      * Register a class alias with the auto-loader.
136:      *
137:      * @param  string  $class
138:      * @param  string  $alias
139:      * @return void
140:      */
141:     public static function alias($class, $alias)
142:     {
143:         static::$aliases[$alias] = $class;
144:     }

146:     /**
147:      * Register directories to be searched as a PSR-0 library.
148:      *
149:      * @param  string|array  $directory
150:      * @return void
151:      */
152:     public static function directories($directory)
153:     {
154:         $directories = static::format($directory);
155: 
156:         static::$directories = array_unique(array_merge(static::$directories, $directories));
157:     }
158: 
159:     /**
160:      * Register underscored "namespaces" to directory mappings.
161:      *
162:      * @param  array  $mappings
163:      * @return void
164:      */
165:     public static function underscored($mappings)
166:     {
167:         static::namespaces($mappings, '_');
168:     }
169: 
170:     /**
171:      * Map namespaces to directories.
172:      *
173:      * @param  array   $mappings
174:      * @param  string  $append
175:      * @return void
176:      */
177:     public static function namespaces($mappings, $append = '\\')
178:     {
179:         foreach ($mappings as $namespace => $directory)
180:         {
181:             // When adding new namespaces to the mappings, we will unset the previously
182:             // mapped value if it existed. This allows previously registered spaces to
183:             // be mapped to new directories on the fly.
184:             $namespace = trim($namespace, $append).$append;
185: 
186:             unset(static::$namespaces[$namespace]);
187: 
188:             $namespaces[$namespace] = head(static::format($directory));
189:         }
190: 
191:         // We'll array_merge the new mappings onto the front of the array so
192:         // derivative namespaces are not always shadowed by their parents.
193:         // For instance, when mappings Laravel\Docs, we don't want the
194:         // main Laravel namespace to always override it.
195:         static::$namespaces = array_merge($namespaces, static::$namespaces);
196:     }
197: 
198:     /**
199:      * Format an array of directories with the proper trailing slashes.
200:      *
201:      * @param  array  $directories
202:      * @return array
203:      */
204:     protected static function format($directories)
205:     {
206:         return array_map(function($directory)
207:         {
208:             return rtrim($directory, DS).DS;
209:         
210:         }, (array) $directories);
211:     }
