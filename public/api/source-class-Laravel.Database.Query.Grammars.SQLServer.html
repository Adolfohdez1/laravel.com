 14:     /**
 15:      * Compile a SQL SELECT statement from a Query instance.
 16:      *
 17:      * @param  Query   $query
 18:      * @return string
 19:      */
 20:     public function select(Query $query)
 21:     {
 22:         $sql = parent::components($query);
 23: 
 24:         // SQL Server does not currently implement an "OFFSET" type keyword, so we
 25:         // actually have to generate the ANSI standard SQL for doing offset like
 26:         // functionality. In the next version of SQL Server, an OFFSET like
 27:         // keyword is included for convenience.
 28:         if ($query->offset > 0)
 29:         {
 30:             return $this->ansi_offset($query, $sql);
 31:         }
 32: 
 33:         // Once all of the clauses have been compiled, we can join them all as
 34:         // one statement. Any segments that are null or an empty string will
 35:         // be removed from the array of clauses before they are imploded.
 36:         return $this->concatenate($sql);
 37:     }

 39:     /**
 40:      * Compile the SELECT clause for a query.
 41:      *
 42:      * @param  Query   $query
 43:      * @return string
 44:      */
 45:     protected function selects(Query $query)
 46:     {
 47:         if ( ! is_null($query->aggregate)) return;
 48: 
 49:         $select = ($query->distinct) ? 'SELECT DISTINCT ' : 'SELECT ';
 50: 
 51:         // Instead of using a "LIMIT" keyword, SQL Server uses the "TOP"
 52:         // keyword within the SELECT statement. So, if we have a limit,
 53:         // we will add it here.
 54:         //
 55:         // We will not add the TOP clause if there is an offset however,
 56:         // since we will have to handle offsets using the ANSI syntax
 57:         // and will need to remove the TOP clause in that situation.
 58:         if ($query->limit > 0 and $query->offset <= 0)
 59:         {
 60:             $select .= 'TOP '.$query->limit.' ';
 61:         }
 62: 
 63:         return $select.$this->columnize($query->selects);
 64:     }
 65: 
 66:     /**
 67:      * Generate the ANSI standard SQL for an offset clause.
 68:      *
 69:      * @param  Query  $query
 70:      * @param  array  $components
 71:      * @return array
 72:      */
 73:     protected function ansi_offset(Query $query, $components)
 74:     {
 75:         // An ORDER BY clause is required to make this offset query
 76:         // work, so if one doesn't exist, we'll just create a dummy
 77:         // clause to satisfy the database.
 78:         if ( ! isset($components['orderings']))
 79:         {
 80:             $components['orderings'] = 'ORDER BY (SELECT 0)';
 81:         }
 82: 
 83:         // We need to add the row number to the query results so we
 84:         // can compare it against the offset and limit values given
 85:         // for the statement. To do that we'll add an expression to
 86:         // the select statement for the row number.
 87:         $orderings = $components['orderings'];
 88: 
 89:         $components['selects'] .= ", ROW_NUMBER() OVER ({$orderings}) AS RowNum";
 90: 
 91:         unset($components['orderings']);
 92: 
 93:         $start = $query->offset + 1;
 94: 
 95:         // Next we need to calculate the constraint that should be
 96:         // placed on the row number to get the correct offset and
 97:         // limit on the query. If a limit has not been set, we'll
 98:         // only add a constraint to handle offset.
 99:         if ($query->limit > 0)
100:         {
101:             $finish = $query->offset + $query->limit;
102: 
103:             $constraint = "BETWEEN {$start} AND {$finish}";
104:         }
105:         else
106:         {
107:             $constraint = ">= {$start}";
108:         }
109: 
110:         // Now, we're finally ready to build the final SQL query.
111:         // We'll create a common table expression with the query
112:         // and then select all of the results from it where the
113:         // row number is between oru given limit and offset.
114:         $sql = $this->concatenate($components);
115: 
116:         return "SELECT * FROM ($sql) AS TempTable WHERE RowNum {$constraint}";
117:     }
