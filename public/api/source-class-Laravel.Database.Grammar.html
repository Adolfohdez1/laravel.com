 30:     /**
 31:      * Wrap a table in keyword identifiers.
 32:      *
 33:      * @param  string  $table
 34:      * @return string
 35:      */
 36:     public function wrap_table($table)
 37:     {
 38:         $prefix = '';
 39: 
 40:         // Tables may be prefixed with a string. This allows developers to
 41:         // prefix tables by application on the same database which may be
 42:         // required in some brown-field situations.
 43:         if (isset($this->connection->config['prefix']))
 44:         {
 45:             $prefix = $this->connection->config['prefix'];
 46:         }
 47: 
 48:         return $this->wrap($prefix.$table);
 49:     }
 50: 
 51:     /**
 52:      * Wrap a value in keyword identifiers.
 53:      *
 54:      * @param  string  $value
 55:      * @return string
 56:      */
 57:     public function wrap($value)
 58:     {
 59:         // Expressions should be injected into the query as raw strings so
 60:         // so we do not want to wrap them in any way. We will just return
 61:         // the string value from the expression to be included.
 62:         if ($value instanceof Expression)
 63:         {
 64:             return $value->get();
 65:         }
 66: 
 67:         // If the value being wrapped contains a column alias, we need to
 68:         // wrap it a little differently as each segment must be wrapped
 69:         // and not the entire string.
 70:         if (strpos(strtolower($value), ' as ') !== false)
 71:         {
 72:             $segments = explode(' ', $value);
 73: 
 74:             return sprintf(
 75:                 '%s AS %s',
 76:                 $this->wrap($segments[0]),
 77:                 $this->wrap($segments[2])
 78:             );
 79:         }
 80: 
 81:         // Since columns may be prefixed with their corresponding table
 82:         // name so as to not make them ambiguous, we will need to wrap
 83:         // the table and the column in keyword identifiers.
 84:         foreach (explode('.', $value) as $segment)
 85:         {
 86:             if ($segment == '*')
 87:             {
 88:                 $wrapped[] = $segment;
 89:             }
 90:             else
 91:             {
 92:                 $wrapped[] = sprintf($this->wrapper, $segment);
 93:             }
 94:         }
 95: 
 96:         return implode('.', $wrapped);
 97:     }
 98: 
 99:     /**
100:      * Create query parameters from an array of values.
101:      *
102:      * <code>
103:      *      Returns "?, ?, ?", which may be used as PDO place-holders
104:      *      $parameters = $grammar->parameterize(array(1, 2, 3));
105:      *
106:      *      // Returns "?, "Taylor"" since an expression is used
107:      *      $parameters = $grammar->parameterize(array(1, DB::raw('Taylor')));
108:      * </code>
109:      *
110:      * @param  array   $values
111:      * @return string
112:      */
113:     final public function parameterize($values)
114:     {
115:         return implode(', ', array_map(array($this, 'parameter'), $values));
116:     }
117: 
118:     /**
119:      * Get the appropriate query parameter string for a value.
120:      *
121:      * <code>
122:      *      // Returns a "?" PDO place-holder
123:      *      $value = $grammar->parameter('Taylor Otwell');
124:      *
125:      *      // Returns "Taylor Otwell" as the raw value of the expression
126:      *      $value = $grammar->parameter(DB::raw('Taylor Otwell'));
127:      * </code>
128:      *
129:      * @param  mixed   $value
130:      * @return string
131:      */
132:     final public function parameter($value)
133:     {
134:         return ($value instanceof Expression) ? $value->get() : '?';
135:     }
136: 
137:     /**
138:      * Create a comma-delimited list of wrapped column names.
139:      *
140:      * <code>
141:      *      // Returns ""Taylor", "Otwell"" when the identifier is quotes
142:      *      $columns = $grammar->columnize(array('Taylor', 'Otwell'));
143:      * </code>
144:      *
145:      * @param  array   $columns
146:      * @return string
147:      */
148:     final public function columnize($columns)
149:     {
150:         return implode(', ', array_map(array($this, 'wrap'), $columns));
151:     }
